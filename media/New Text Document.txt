<script>
    const parrot = document.getElementById("parrot");

    // Start at a random location
    function randomPosition() {
        const x = Math.random() * (window.innerWidth - 150);
        const y = Math.random() * (window.innerHeight - 150);
        return { x, y };
    }

    // Move the parrot to random places on the screen
    function moveParrot() {
        const pos = randomPosition();
        parrot.style.transform = `translate(${pos.x}px, ${pos.y}px)`;
    }

    // Initial placement
    moveParrot();

    // Move every 2.8 seconds (hover + fly)
    setInterval(moveParrot, 2800);
</script>
